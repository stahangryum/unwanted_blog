Problem 1

Solution I - for 

px = c()
py = c()
px[1] = 1
py[1] = 1

for (n in 1:2**12){
    if ((px[n]==10)&(py[n]==2^10)){
        print(n)
        break
    }else{
        if (py[n] < 2**(px[n])){
            px[n+1] = px[n]
            py[n+1] = py[n] + 1
        }else{
            px[n+1] = px[n] + 1
            py[n+1] = 1
                   
        }
    }
}

[1] 2046

$n = 2046 = 2048 - 2 = 2^{11} - 2$
